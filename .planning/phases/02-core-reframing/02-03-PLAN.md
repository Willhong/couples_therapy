```typescript
  updateGoals: async (goals: Partial<UserGoals>): Promise<UserGoals> => {
    const response = await api.put('/onboarding/goals/', goals);
```

**useOnboardingProgress.ts:**
```typescript
import { useState, useEffect } from 'react';
import { onboardingApi } from '../services/onboardingApi';
import { OnboardingStatus, UserProfile, UserGoals } from '../types';

export function useOnboardingProgress() {
  const [status, setStatus] = useState<OnboardingStatus | null>(null);
  const [loading, setLoading] = useState(true);
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const fetchStatus = async () => {
    try {
      const data = await onboardingApi.getStatus();
      setStatus(data);
    } catch (err) {
      setError('상태를 불러올 수 없습니다');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchStatus();
  }, []);

  const submitOnboarding = async (profile: UserProfile, goals: UserGoals) => {
    setSubmitting(true);
    setError(null);
    try {
      await onboardingApi.saveProfile(profile);
      await onboardingApi.saveGoals(goals);
      await fetchStatus();
      return true;
    } catch (err: any) {
      setError(err.response?.data?.error || '저장에 실패했습니다');
      return false;
    } finally {
      setSubmitting(false);
    }
  };

  return {
    status,
    loading,
    submitting,
    error,
    submitOnboarding,
    refetch: fetchStatus,
  };
}
```

**QuestionnaireWizard.tsx:**
```typescript
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native';
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';
import { useRouter } from 'expo-router';

import { ProgressBar } from '@/components/onboarding/ProgressBar';
import { AttachmentStyleStep } from './AttachmentStyleStep';
import { ConflictStyleStep } from './ConflictStyleStep';
import { GoalSelectionStep } from './GoalSelectionStep';
import { useOnboardingProgress } from '../hooks/useOnboardingProgress';

const onboardingSchema = z.object({
  attachmentAnxiety: z.number().min(1).max(5),
  attachmentAvoidance: z.number().min(1).max(5),
  conflictStyle: z.enum(['avoid', 'confront', 'collaborate', 'compromise']),
  communicationFrequency: z.enum(['daily', 'weekly', 'rarely']),
  primaryGoal: z.enum(['prevention', 'improvement', 'crisis']),
  focusAreas: z.array(z.string()).min(1, '최소 1개를 선택해주세요').max(3, '최대 3개까지 선택 가능합니다'),
});

type OnboardingData = z.infer<typeof onboardingSchema>;

const STEPS = [
  { id: 'attachment', title: '소통 스타일' },
  { id: 'conflict', title: '갈등 대응' },
  { id: 'goals', title: '목표 설정' },
];

export function QuestionnaireWizard() {
  const [currentStep, setCurrentStep] = useState(0);
  const router = useRouter();
  const { submitOnboarding, submitting, error } = useOnboardingProgress();

  const { control, handleSubmit, formState: { errors }, trigger } = useForm<OnboardingData>({
    resolver: zodResolver(onboardingSchema),
    defaultValues: {
      attachmentAnxiety: 3,
      attachmentAvoidance: 3,
      focusAreas: [],
    },
  });

  const goNext = async () => {
    // Validate current step fields
    const stepFields: Record<number, (keyof OnboardingData)[]> = {
      0: ['attachmentAnxiety', 'attachmentAvoidance'],
      1: ['conflictStyle', 'communicationFrequency'],
      2: ['primaryGoal', 'focusAreas'],
    };

    const isValid = await trigger(stepFields[currentStep]);
    if (isValid) {
      setCurrentStep(prev => Math.min(prev + 1, STEPS.length - 1));
    }
  };

  const goBack = () => setCurrentStep(prev => Math.max(prev - 1, 0));

  const onSubmit = async (data: OnboardingData) => {
    const profile = {
      attachment_anxiety: data.attachmentAnxiety,
      attachment_avoidance: data.attachmentAvoidance,
      conflict_style: data.conflictStyle,
      communication_frequency: data.communicationFrequency,
    };
    const goals = {
      primary_goal: data.primaryGoal,
      focus_areas: data.focusAreas,
    };

    const success = await submitOnboarding(profile, goals);
    if (success) {
      router.replace('/onboarding/partner-link');
    }
  };

  return (
    <View style={styles.container}>
      <ProgressBar current={currentStep + 1} total={STEPS.length} />
      <Text style={styles.stepTitle}>{STEPS[currentStep].title}</Text>

      {currentStep === 0 && <AttachmentStyleStep control={control} errors={errors} />}
      {currentStep === 1 && <ConflictStyleStep control={control} errors={errors} />}
      {currentStep === 2 && <GoalSelectionStep control={control} errors={errors} />}

      {error && <Text style={styles.errorText}>{error}</Text>}

      <View style={styles.navigation}>
        {currentStep > 0 && (
          <TouchableOpacity style={styles.backButton} onPress={goBack}>
            <Text style={styles.backText}>이전</Text>
          </TouchableOpacity>
        )}
        {currentStep < STEPS.length - 1 ? (
          <TouchableOpacity style={styles.nextButton} onPress={goNext}>
            <Text style={styles.nextText}>다음</Text>
          </TouchableOpacity>
        ) : (
          <TouchableOpacity
            style={[styles.nextButton, submitting && styles.buttonDisabled]}
            onPress={handleSubmit(onSubmit)}
            disabled={submitting}
          >
            {submitting ? (
              <ActivityIndicator color="#fff" />
            ) : (
              <Text style={styles.nextText}>완료</Text>
            )}
          </TouchableOpacity>
        )}
      </View>
    </View>
  );
}
```
