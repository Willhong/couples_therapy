"""
LangChain LLM service with provider abstraction.
Supports OpenAI, Anthropic, and Google providers.
"""
import json
from typing import Optional
from django.conf import settings

from langchain_core.language_models import BaseChatModel


def get_chat_model(for_summarization: bool = False) -> BaseChatModel:
    """
    Factory function to get the appropriate LangChain chat model
    based on settings.LLM_PROVIDER.

    Args:
        for_summarization: Use smaller/cheaper model for summarization

    Returns:
        LangChain ChatModel instance
    """
    provider = settings.LLM_PROVIDER
    model_type = "summarization" if for_summarization else "chat"
    model_name = settings.LLM_MODELS[provider][model_type]

    common_kwargs = {
        "temperature": settings.LLM_TEMPERATURE,
        "max_tokens": settings.LLM_MAX_TOKENS if not for_summarization else 300,
    }

    if provider == "openai":
        from langchain_openai import ChatOpenAI
        return ChatOpenAI(model=model_name, **common_kwargs)

    elif provider == "anthropic":
        from langchain_anthropic import ChatAnthropic
        return ChatAnthropic(model=model_name, **common_kwargs)

    elif provider == "google":
        from langchain_google_genai import ChatGoogleGenerativeAI
        return ChatGoogleGenerativeAI(model=model_name, **common_kwargs)

    else:
        raise ValueError(f"Unsupported LLM provider: {provider}")


def parse_json_response(response_text: str) -> dict:
    """
    Parse LLM's JSON response, handling markdown code blocks.
    """
    try:
        # Remove markdown code blocks if present
        text = response_text.strip()
        if text.startswith("```json"):
            text = text[7:]
        elif text.startswith("```"):
            text = text[3:]
        if text.endswith("```"):
            text = text[:-3]

        # Find JSON object
        json_start = text.find('{')
        json_end = text.rfind('}') + 1
        if json_start >= 0 and json_end > json_start:
            return json.loads(text[json_start:json_end])
    except json.JSONDecodeError:
        pass
    return {}


def parse_json_array(response_text: str) -> list:
    """
    Parse LLM's JSON array response.
    """
    try:
        text = response_text.strip()
        if text.startswith("```"):
            text = text.split("```")[1]
            if text.startswith("json"):
                text = text[4:]

        json_start = text.find('[')
        json_end = text.rfind(']') + 1
        if json_start >= 0 and json_end > json_start:
            return json.loads(text[json_start:json_end])
    except json.JSONDecodeError:
        pass
    return []


def get_provider_info() -> dict:
    """
    Returns information about current LLM configuration.
    """
    provider = settings.LLM_PROVIDER
    return {
        "provider": provider,
        "chat_model": settings.LLM_MODELS[provider]["chat"],
        "summarization_model": settings.LLM_MODELS[provider]["summarization"],
        "max_tokens": settings.LLM_MAX_TOKENS,
        "temperature": settings.LLM_TEMPERATURE,
    }
```

"""
LangGraph reframing pipeline for couples therapy.

Graph flow:
  input -> safety_check -> [if abuse: safety_response]
                        -> [else: acknowledge -> analyze -> suggest -> combine]
"""
import json
from typing import TypedDict, Annotated, Literal
from operator import add

from langgraph.graph import StateGraph, END
from langchain_core.messages import HumanMessage, SystemMessage

from apps.chat.services.llm_service import get_chat_model, parse_json_response, parse_json_array
from apps.chat.prompts import (
    ACKNOWLEDGE_PROMPT,
    ANALYZE_PROMPT,
    SUGGEST_PROMPT,
    SAFETY_CHECK_PROMPT,
)


class ReframingState(TypedDict):
    """State passed through the reframing graph."""
    # Input
    user_message: str
    conversation_context: str

    # Safety check
    abuse_detected: bool
    abuse_info: dict

    # Processing results
    acknowledgment: str
    analysis: dict
    suggestions: list

    # Final output
    reframing_result: dict
    error: str


def safety_check_node(state: ReframingState) -> ReframingState:
    """Check for abuse patterns in user message."""
    model = get_chat_model()

    response = model.invoke([
        SystemMessage(content=SAFETY_CHECK_PROMPT),
        HumanMessage(content=state["user_message"])
    ])

    result = parse_json_response(response.content)

    return {
        **state,
        "abuse_detected": result.get("abuse_detected", False),
        "abuse_info": result,
    }


def acknowledge_node(state: ReframingState) -> ReframingState:
    """Acknowledge user's emotions."""
    model = get_chat_model()

    response = model.invoke([
        SystemMessage(content=ACKNOWLEDGE_PROMPT),
        HumanMessage(content=f"대화 맥락:\n{state['conversation_context']}\n\n현재 메시지:\n{state['user_message']}")
    ])

    return {
        **state,
        "acknowledgment": response.content,
    }


def analyze_node(state: ReframingState) -> ReframingState:
    """Analyze perspectives from both sides."""
    model = get_chat_model()

    response = model.invoke([
        SystemMessage(content=ANALYZE_PROMPT),
        HumanMessage(content=f"대화 맥락:\n{state['conversation_context']}\n\n현재 메시지:\n{state['user_message']}")
    ])

    analysis = parse_json_response(response.content)

    return {
        **state,
        "analysis": analysis,
    }


def suggest_node(state: ReframingState) -> ReframingState:
    """Generate actionable suggestions."""
    model = get_chat_model()

    # Include analysis context for better suggestions
    analysis_context = json.dumps(state.get("analysis", {}), ensure_ascii=False)

    response = model.invoke([
        SystemMessage(content=SUGGEST_PROMPT),
        HumanMessage(content=f"상황 분석:\n{analysis_context}\n\n원본 메시지:\n{state['user_message']}")
    ])

    suggestions = parse_json_array(response.content)
    if not suggestions:
        suggestions = [response.content]

    return {
        **state,
        "suggestions": suggestions,
    }


def combine_node(state: ReframingState) -> ReframingState:
    """Combine all results into final reframing response."""
    analysis = state.get("analysis", {})

    result = {
        "acknowledgment": state.get("acknowledgment", ""),
        "what_you_said": analysis.get("what_you_said", ""),
        "how_they_heard": analysis.get("how_they_heard", ""),
        "how_you_heard_them": analysis.get("how_you_heard_them", ""),
        "why_the_gap": analysis.get("why_the_gap", ""),

    """Route based on abuse detection."""
    if state.get("abuse_detected"):
        return "safety_response"
    return "acknowledge"


# Build the graph
def build_reframing_graph() -> StateGraph:
    """Build and compile the reframing graph."""
    graph = StateGraph(ReframingState)

    # Add nodes
    graph.add_node("safety_check", safety_check_node)
    graph.add_node("safety_response", safety_response_node)
    graph.add_node("acknowledge", acknowledge_node)
    graph.add_node("analyze", analyze_node)
    graph.add_node("suggest", suggest_node)
    graph.add_node("combine", combine_node)

    # Set entry point
    graph.set_entry_point("safety_check")

    # Add conditional edge after safety check
    graph.add_conditional_edges(
        "safety_check",
        route_after_safety,
        {
            "safety_response": "safety_response",
            "acknowledge": "acknowledge",
        }
    )

    # Safety response goes to end
    graph.add_edge("safety_response", END)

    # Normal flow: acknowledge -> analyze -> suggest -> combine
    graph.add_edge("acknowledge", "analyze")
    graph.add_edge("analyze", "suggest")
    graph.add_edge("suggest", "combine")
    graph.add_edge("combine", END)

    return graph.compile()

"""
Conversation context manager with rolling summarization.
Uses LangChain for provider-agnostic LLM calls.
"""
from typing import Optional
from langchain_core.messages import HumanMessage

from apps.chat.services.llm_service import get_chat_model
from apps.chat.prompts import SUMMARIZATION_PROMPT


class ConversationContextManager:
    """
    Manages conversation context using rolling summarization.
    Keeps recent messages verbatim + summarized older context.
    """

    MAX_RECENT_MESSAGES = 10  # Keep last 10 messages verbatim
    SUMMARY_TRIGGER_COUNT = 20  # Summarize when exceeding this

    def __init__(self, conversation_id):
        self.conversation_id = conversation_id
        self._cached_summary = None

    def get_context_for_ai(self) -> str:
        """
        Returns optimized context string for LLM API.
        Format: [Summary of older messages] + [Recent messages verbatim]
        """
        from apps.chat.models import Message, ConversationSummary

        messages = Message.objects.filter(
            conversation_id=self.conversation_id
        ).order_by('-created_at')

        total_count = messages.count()

        if total_count <= self.MAX_RECENT_MESSAGES:
            # Few messages: return all verbatim
            return self._format_messages(list(reversed(list(messages))))

        # Get recent messages verbatim
        recent_messages = list(messages[:self.MAX_RECENT_MESSAGES])
        recent_messages.reverse()

        # Get or create summary of older messages
        summary = self._get_or_create_summary()

        context_parts = []
        if summary:
            context_parts.append(f"[이전 대화 요약]\n{summary}\n")
        context_parts.append("[최근 대화]")
        context_parts.append(self._format_messages(recent_messages))

        return "\n".join(context_parts)

    def _get_or_create_summary(self) -> str:
        """Get cached summary or generate new one."""
        from apps.chat.models import Message, ConversationSummary

        # Check for existing valid summary
        existing = ConversationSummary.objects.filter(
            conversation_id=self.conversation_id
        ).order_by('-created_at').first()

        if existing and self._is_summary_current(existing):
            return existing.summary_text

        # Generate new summary
        all_messages = Message.objects.filter(
            conversation_id=self.conversation_id
        ).order_by('created_at')

        older_messages = list(all_messages)[:-self.MAX_RECENT_MESSAGES]

        if not older_messages:
            return ""

        summary_text = self._generate_summary(older_messages)

        ConversationSummary.objects.create(
            conversation_id=self.conversation_id,
            summary_text=summary_text,
            message_count=len(older_messages)
        )

        return summary_text

    def _generate_summary(self, messages) -> str:
        """Use LangChain to summarize older messages."""
        formatted = self._format_messages(messages)

        # Get summarization model (smaller/cheaper)
        model = get_chat_model(for_summarization=True)

        prompt_content = SUMMARIZATION_PROMPT.format(conversation=formatted)
        response = model.invoke([HumanMessage(content=prompt_content)])

        return response.content

    def _format_messages(self, messages) -> str:
        """Format messages for LLM context."""
        lines = []
        for msg in messages:
            role = "나" if msg.role == 'user' else "AI"
            lines.append(f"{role}: {msg.content}")
        return "\n".join(lines)

    def _is_summary_current(self, summary) -> bool:
        """Check if summary is still valid."""
        from apps.chat.models import Message
        current_count = Message.objects.filter(
            conversation_id=self.conversation_id
        ).count()
        # Summary is current if message count hasn't grown too much
        return current_count - summary.message_count < self.SUMMARY_TRIGGER_COUNT

@api_view(['POST'])
@permission_classes([IsAuthenticated])
def reframe_message(request):
    """
    Process user message through LangGraph reframing pipeline.

    Request body:
    {
        "conversation_id": "uuid",
        "message": "user's message text"
    }

    Response: Reframing result JSON
    """
    conversation_id = request.data.get('conversation_id')
    message = request.data.get('message')

    if not conversation_id or not message:
        return Response(
            {"error": "conversation_id와 message가 필요합니다"},
            status=400
        )

    # Verify user owns conversation
    try:
        conversation = Conversation.objects.get(
            id=conversation_id,
            user=request.user
        )
    except Conversation.DoesNotExist:
        return Response(
            {"error": "대화를 찾을 수 없습니다"},
            status=404
        )

    # Save user message
    user_msg = Message.objects.create(
        conversation=conversation,
        role='user',
        content=message
    )

    # Get conversation context
    context_manager = ConversationContextManager(str(conversation_id))
    context = context_manager.get_context_for_ai()

    # Run LangGraph pipeline
    result = run_reframing_pipeline_sync(message, context)

    if "error" in result:
        return Response({"error": result["error"]}, status=500)

    # Save assistant message with reframing data
    assistant_msg = Message.objects.create(
        conversation=conversation,
        role='assistant',
        content=json.dumps(result, ensure_ascii=False),
        has_reframing=True,
        reframing_data=result
    )

    # Update conversation title if first exchange
    if not conversation.title:
        conversation.title = message[:50] + ("..." if len(message) > 50 else "")
        conversation.save()

    return Response({
        "message_id": str(assistant_msg.id),
        "reframing": result
    }, status=201)

            conversation.save()
