    ```python
    @api_view(['POST'])
    @permission_classes([IsAuthenticated])
    def end_conversation(request, conversation_id):
        """Mark conversation as ended and trigger analysis evaluation.

        Called when user navigates away from chat screen.
        """
        from apps.chat.models import Conversation

        try:
            conversation = Conversation.objects.get(
                pk=conversation_id, user=request.user,
            )
        except Conversation.DoesNotExist:
            return Response(
                {'detail': '대화를 찾을 수 없습니다.'},
                status=status.HTTP_404_NOT_FOUND,
            )

        # Fire analysis trigger (if accumulative therapy enabled)
        from django.conf import settings as django_settings
        if getattr(django_settings, 'ACCUMULATIVE_THERAPY_ENABLED', False):
            try:
                from apps.intelligence.tasks import on_conversation_ended
                on_conversation_ended.delay(str(conversation.id), str(request.user.id))
            except Exception as e:
                logger.warning("Failed to queue conversation end trigger: %s", e)

        return Response({'detail': '대화가 종료되었습니다.'}, status=status.HTTP_200_OK)
    ```


    With lazy compilation (same pattern already used in analysis_graph.py):
    ```python
    # Lazy compilation (avoid import-time side effects)
    _compiled_graph = None


    def _get_compiled_graph():
        global _compiled_graph
        if _compiled_graph is None:
            _compiled_graph = _build_graph().compile()
        return _compiled_graph
    ```

    ```python
    result_state = await _compiled_graph.ainvoke(initial_state)
    ```

    Replace with:
    ```python
    result_state = await _get_compiled_graph().ainvoke(initial_state)
    ```

    This ensures the graph is only compiled on first use, not at import time. The `.ainvoke()` is correct here because `run_chat_agent_pipeline` is an async function called via `async_to_sync` from views.
