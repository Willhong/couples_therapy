Optimize the useWaveform hook to reduce unnecessary state updates:

1. Use throttling to batch updates - only update state every 150ms instead of every metering value:
   ```tsx
   import { useState, useCallback, useRef } from 'react';

   export function useWaveform(maxBars: number = 50): UseWaveformReturn {
     const [waveformData, setWaveformData] = useState<number[]>([]);
     const dataRef = useRef<number[]>([]);
     const lastUpdateRef = useRef<number>(0);
     const pendingUpdateRef = useRef<ReturnType<typeof setTimeout> | null>(null);

     const THROTTLE_MS = 150; // Batch updates every 150ms

     const flushUpdate = useCallback(() => {
       setWaveformData([...dataRef.current]);
       pendingUpdateRef.current = null;
     }, []);

     const addMeteringValue = useCallback(
       (value: number) => {
         const clamped = Math.max(0, Math.min(1, value));
         const newData = [...dataRef.current, clamped];

         if (newData.length > maxBars) {
           newData.splice(0, newData.length - maxBars);
         }

         dataRef.current = newData;

         const now = Date.now();
         if (now - lastUpdateRef.current >= THROTTLE_MS) {
           // Enough time has passed, update immediately
           lastUpdateRef.current = now;
           setWaveformData([...dataRef.current]);
         } else if (!pendingUpdateRef.current) {
           // Schedule an update
           pendingUpdateRef.current = setTimeout(() => {
             lastUpdateRef.current = Date.now();
             flushUpdate();
           }, THROTTLE_MS - (now - lastUpdateRef.current));
         }
       },
       [maxBars, flushUpdate]
     );

     const reset = useCallback(() => {
       if (pendingUpdateRef.current) {
         clearTimeout(pendingUpdateRef.current);
         pendingUpdateRef.current = null;
       }
       dataRef.current = [];
       setWaveformData([]);
     }, []);

     return { waveformData, addMeteringValue, reset };
   }
   ```
2. This throttling ensures we don't trigger a re-render for every 100ms metering update

   ```tsx
   import React, { memo } from 'react';

   function WaveformVisualizerComponent({
     waveformData,
     isRecording,
   }: Props): React.ReactElement {
     // ... existing implementation
   }

   export const WaveformVisualizer = memo(WaveformVisualizerComponent, (prev, next) => {
     // Only re-render if isRecording changes or data length changes significantly
     if (prev.isRecording !== next.isRecording) return false;
     if (prev.waveformData.length !== next.waveformData.length) return false;
     // Compare last few values to detect actual changes
     const len = prev.waveformData.length;
     if (len === 0) return true;
     return prev.waveformData[len - 1] === next.waveformData[len - 1];
   });
   ```
