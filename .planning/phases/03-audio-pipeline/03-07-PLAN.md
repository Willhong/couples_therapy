       ```typescript
       export interface DashboardData {
         total_sessions: number;
         top_topics: { name: string; category: string; count: number }[];
         top_triggers: { text: string; count: number }[];
         escalation_trend: { week: string; avg_score: number }[];
         conflict_frequency: { week: string; count: number }[];
       }

       export interface WeeklySummaryData {
         id: string;
         period_start: string;
         period_end: string;
         session_count: number;
         top_topics: { topic: string; count: number }[];
         trigger_frequency: Record<string, number>;
         trend_text: string;
         escalation_trend: 'improving' | 'stable' | 'worsening';
         created_at: string;
       }

       export interface SessionInsight {
         id: string;
         conversation_id: string;
         trigger_phrases: { text: string; context: string }[];
         recurring_topics: { name: string; category: string; prior_count: number }[];
         escalation_score: number;
         ai_summary: string;
       }
       ```
